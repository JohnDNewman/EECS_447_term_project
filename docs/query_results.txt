    itemDetails.itemType,
    SUM(p.paymentValue) AS total_revenue
FROM 
    payments p
JOIN 
    borrows b ON p.userID = b.userID

        SELECT itemID, 'Book' AS itemType FROM books
        UNION ALL
        SELECT itemID, 'Magazine' AS itemType FROM magazines

    AND MONTH(p.datePayed) = 4
